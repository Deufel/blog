üîß Manual Inspector Tools

                
Open browser console and try these commands:

                
// Find all elements with data-* attributes
document.querySelectorAll('[data-]')

// Find all elements with IDs
document.querySelectorAll('[id]')

// Find all Datastar-specific attributes
Array.from(document.querySelectorAll('*')).filter(el =>
    Array.from(el.attributes).some(attr =>
        attr.name.startsWith('data-') &&
        (attr.name.includes('signal') || attr.name.includes('text') || attr.name.includes('show'))
    )
)

// Get all unique data-* attribute names
Array.from(new Set(
    Array.from(document.querySelectorAll('[data-]'))
        .flatMap(el => Array.from(el.attributes))
        .filter(attr => attr.name.startsWith('data-'))
        .map(attr => attr.name)
)).sort()
